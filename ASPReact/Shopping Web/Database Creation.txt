CREATE TABLE "Products" (
	"Id"	INTEGER NOT NULL UNIQUE,
	"Name"	TEXT NOT NULL,
	"Price"	NUMERIC NOT NULL,
	"Title"	TEXT,
	"Image"	TEXT,
	"URL"	TEXT,
	"Description"	TEXT,
	PRIMARY KEY("Id" AUTOINCREMENT)
)

CREATE TABLE "User" (
	"ID"	INTEGER NOT NULL UNIQUE,
	"Name"	TEXT NOT NULL,
	"Email"	TEXT NOT NULL UNIQUE,
	"Password"	INTEGER NOT NULL,
	PRIMARY KEY("ID" AUTOINCREMENT)
);
